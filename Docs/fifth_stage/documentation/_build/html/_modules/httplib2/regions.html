[docs]    def get_all_stations(self):
        """
        :return: sorted list of main cities
        """
        while self.BREAKPOINTS[0] != self.startpos:
            self.BREAKPOINTS.append(self.BREAKPOINTS[0])
            del self.BREAKPOINTS[0]
        self.get_stations(self.BREAKPOINTS)
        return self.array()
